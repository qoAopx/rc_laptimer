/*
 * ==============================================================================
 * RC Car Lap Timer (Hose Type) - LCD2004 (20x4) Version
 * ==============================================================================
 * [プログラム概要]
 * チューブ内の空気圧変化を微分（変化率）解析し、マシンの通過を検知します。
 * 計測データは I2C接続の 20x4 液晶ディスプレイに表示されます。
 * * [ピンアサイン]
 * - D2 : HX710B OUT (データ入力)
 * - D3 : HX710B SCK (同期クロック)
 * - D4 : 圧電ブザー (報知用)
 * - A0 : 可変抵抗 (感度調整しきい値用)
 * - A4 : I2C SDA (液晶データ)
 * - A5 : I2C SCL (液晶クロック)
 * * [アルゴリズム詳説]
 * 1. 移動平均: ノイズを除去し、信号を滑らかにします。
 * 2. 微分処理: 気温変化等の「ゆっくりした変動」を無視し、「踏んだ瞬間」を捉えます。
 * 3. 二重カウント防止: 一度検知すると、圧力が抜けるまで再検知をロックします。
 * ==============================================================================
 */

#include <Wire.h>
#include <LiquidCrystal_I2C.h>

// 液晶の初期設定：アドレス 0x27 (または0x3F), 20列, 4行
LiquidCrystal_I2C lcd(0x27, 20, 4);

// --- ピンの定義 ---
const int HX710B_OUT = 2; // センサーデータ出力
const int HX710B_SCK = 3; // センサー同期クロック
const int VR_PIN = A0;    // 感度調整ボリューム
const int BUZZER_PIN = 4; // 報知用ブザー

// --- グローバル変数 ---
const int AVG_SIZE = 4;           // 移動平均のサンプル数
long readings[AVG_SIZE];          // 平均化用のバッファ
int readIndex = 0;                // バッファ管理インデックス
long total = 0;                   // 合計値

unsigned long lastLapTime = 0;    // 前回ラップ検知時刻(ms)
const unsigned long DEAD_TIME = 10000; // 検知後の無反応時間(ms) 10秒
float currentLap = 0.0;           // 最新ラップ
float bestTime = 99999.99;          // ベストタイム
float prevLap = 0.0;              // 前回のラップ
bool isFirstRun = true;           // スタート判定フラグ
bool isTriggered = false;         // 踏み込み中フラグ

/*
 * formatTime: 秒数を「00:00.00」形式の文字列に変換します。
 * sprintfを使用して、分と秒の桁数を揃えて美しく表示します。
 */
String formatTime(float totalSeconds) {
  if (totalSeconds >= 3600 || totalSeconds <= 0) return "00:00.00";

  // sprintfは小数点%fは使えないので整数に変換
  int totalCentiSec = (int)(totalSeconds * 100 + 0.5);  // 100倍してセンチ秒 ｓ四捨五入
  int m = totalCentiSec / 6000;
  int s = (totalCentiSec % 6000) / 100;
  int cs = totalCentiSec % 100;

  char buf[16];
  sprintf(buf, "%02d:%02d.%02d", m, s, cs);
  return String(buf);
}

/*
 * updateLCD: 液晶ディスプレイの描画を更新します。
 * 1行目には、調整を容易にするためリアルタイムの圧力変化(P)としきい値(TH)を出します。
 */
void updateLCD(float lapTime, float bestTime, long currentSlope, int threshold) {
  // 1行目：モニタリング情報
  lcd.setCursor(0, 0);
  lcd.print("P:"); lcd.print(currentSlope);
  lcd.print(" TH:"); lcd.print(threshold);
  lcd.print("      ");

  // 2行目：ベストタイム
  lcd.setCursor(0, 1);
  lcd.print("BEST : ");
  lcd.print(formatTime(bestTime));

  // 3行目：最新ラップ
  lcd.setCursor(0, 2);
  lcd.print("CURR : ");
  lcd.print(formatTime(lapTime));

  // 4行目：前回のラップ
  lcd.setCursor(0, 3);
  lcd.print("PREV : ");
  lcd.print(formatTime(prevLap));
}

/*
 * readHX710B: HX710Bセンサーから24bitデータを取得します。
 * クロックタイミングを制御してデータを読み出し、long型に拡張して返します。
 */
long readHX710B() {
  while (digitalRead(HX710B_OUT) == HIGH);
  unsigned long value = 0;
  for (int i = 0; i < 24; i++) {
    digitalWrite(HX710B_SCK, HIGH);
    value = (value << 1) | digitalRead(HX710B_OUT);
    digitalWrite(HX710B_SCK, LOW);
  }
  digitalWrite(HX710B_SCK, HIGH);
  digitalWrite(HX710B_SCK, LOW);
  if (value & 0x800000) value |= 0xFF000000;
  return (long)value;
}

/*
 * beep: 指定回数だけブザーを鳴らします。
 */
void beep(int count) {
  for(int i=0; i<count; i++) {
    tone(BUZZER_PIN, 3400, 20); // 100ms
  }
}

/*
 * fanfare: ブザーでファンファーレを鳴らします。
 */
void fanfare() {
  tone(BUZZER_PIN, 523, 180);  // C5
  delay(200);
  tone(BUZZER_PIN, 659, 180);  // E5
  delay(200);
  tone(BUZZER_PIN, 784, 180);  // G5
  delay(200);
  tone(BUZZER_PIN, 1047, 260); // C6
  delay(280);
  tone(BUZZER_PIN, 784, 160);  // G5
  delay(180);
  tone(BUZZER_PIN, 880, 160);  // A5
  delay(180);
  tone(BUZZER_PIN, 988, 180);  // B5
  delay(200);
  tone(BUZZER_PIN, 1047, 400); // C6 (フィニッシュ)
  delay(450);

  noTone(BUZZER_PIN);
}

/*
 * bootSound: ブザーでオープニングを鳴らします。
 */
void bootSound() {
  tone(BUZZER_PIN, 880, 180);   // A5
  delay(200);
  tone(BUZZER_PIN, 988, 180);   // B5
  delay(200);
  tone(BUZZER_PIN, 1109, 180);  // C#6
  delay(200);
  tone(BUZZER_PIN, 1319, 220);  // E6
  delay(240);
  tone(BUZZER_PIN, 1175, 160);  // D6
  delay(180);
  tone(BUZZER_PIN, 1109, 160);  // C#6
  delay(180);
  tone(BUZZER_PIN, 988, 200);   // B5
  delay(220);
  tone(BUZZER_PIN, 1319, 400);  // E6（決め）
  delay(450);

  noTone(BUZZER_PIN);
}

/*
 * setup: 電源投入時の初期化。
 * センサーの初期値を取得し、移動平均のベースを作ります。
 */
void setup() {
  pinMode(HX710B_OUT, INPUT);
  pinMode(HX710B_SCK, OUTPUT);
  pinMode(BUZZER_PIN, OUTPUT);
  Serial.begin(115200);
  lcd.init();
  lcd.clear();
  lcd.backlight();
  lcd.setCursor(0, 0);
  lcd.print("LAPTIMER STARTING...");
  Serial.println("LAPTIMER STARTING...");
  bootSound();

  long initialPressure = readHX710B();
  for (int i = 0; i < AVG_SIZE; i++) {
    readings[i] = initialPressure;
    total += initialPressure;
  }

  beep(1);
  lcd.clear();
}

/*
 * loop: メイン処理。
 * 微分値（slope）がボリュームで設定したしきい値を超えたらラップを記録します。
 */
void loop() {
  long valHX7100 = readHX710B();

  /* debug センサー偽造*/
  randomSeed(millis());
  valHX7100 = random(0,1001) - 500;
  if (random(0,1000) > 997) {
    randomSeed(millis());
    valHX7100 = valHX7100 * random(30);
  }

  // 1. 移動平均フィルタ
  total -= readings[readIndex];
  readings[readIndex] = valHX7100;
  total += readings[readIndex];
  readIndex = (readIndex + 1) % AVG_SIZE;
  long averagePressure = total / AVG_SIZE;

  // 2. 微分（前ステップとの差分）
  static long prevAverage = averagePressure;
  long slope = averagePressure - prevAverage;

  // 3. ボリューム値からしきい値を計算 3.3Vでmax665
  int threshold = map(analogRead(VR_PIN), 0, 700, 100, 9999);
  /*
  Serial.print("VR:");
  Serial.print(analogRead(VR_PIN));
  Serial.print(" TH:");
  Serial.println((threshold));
  */

  //if (threshold > 3000) beep(1);
  bool logFlag = false;
  //bool logFlag = true;
  if (logFlag) {
    Serial.print("+TH:");
    Serial.print(threshold);
    Serial.print(" ");
    Serial.print("VAL:");
    Serial.print(slope);
    Serial.print(" ");
    Serial.print("-TH:");
    Serial.println(-threshold);
  }

  // 液晶の定期更新（0.25秒おき）
  static unsigned long lastLcdUpdate = 0;
  if (millis() - lastLcdUpdate > 250) {
    updateLCD(currentLap, bestTime, slope, threshold);
    lastLcdUpdate = millis();
  }

  // 4. ラップ検知ロジック
  if (!isTriggered) {
    if (slope > threshold && (millis() - lastLapTime) > DEAD_TIME) {
      unsigned long now = millis();
      isTriggered = true; // チャタリング防止ロック

      if (!isFirstRun) {
        prevLap = currentLap;
        currentLap = (now - lastLapTime) / 1000.0;

        bool isBest = false;
        if (currentLap < bestTime) {
          bestTime = currentLap;
          isBest = true;
        }

        // ベストラップなら３回、普通のラップは１回
        if (isBest) {
          //beep(3);
          fanfare();
          Serial.print("BEST:");
          Serial.println((bestTime));
          //Serial.print(" ");
          //Serial.println(formatTime(bestTime));
        } else {
          beep(1);
          Serial.print(" Lap:");
          Serial.println((currentLap));
          //Serial.print(" ");
          //Serial.println(formatTime(currentLap));
        }

      } else {
        // はじめにラップを開始したとき
        isFirstRun = false;
        beep(2);
        Serial.println("FirstRun");
      }
      lastLapTime = now;
    }
  } else {
    // 圧力が解放されたら（負の微分）ロック解除
    if (slope < -(threshold / 2)) {
      isTriggered = false;
    }
  }

  prevAverage = averagePressure;
}
